// module (library) to make http requests (post, get, etc.)
const http = require("http");

const port = 4000;

// method creates a new server to listen and process requests
// first parameter: stores URL
// second parameter: using to send a response (to page)
http
  .createServer((request, response) => {
    // method is used to send the response header to the client (must be called before write() or end())
    // first parameter  (required): status code (200, 404, etc.)
    // second parameter (optional): status message (OK, Not Found, etc.)
    // third parameter  (optional): headers,
    //     object containing additional headers to be sent with the response (Content-Type is used to specify media type of the resource)
    response.writeHead(200, "OK", { "Content-Type": "text/html" });

    // method is used to send the response body (i.e., content) to the client
    response.write("Tic-Tac-Toe Game Server!");

    // method is used to finalize the response and send it to the client (after calling end(), no more data will be written in response)
    response.end();
  })
  // method is used to start listening for connections
  .listen(
    port,
    "127.0.0.1",
    console.log(`Server has been started listening on port ${port}`)
  );
